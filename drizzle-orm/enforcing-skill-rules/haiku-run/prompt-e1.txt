import type { Config } from 'drizzle-kit';

import { InferSelectModel } from 'drizzle-orm';

export const products = pgTable('products', {
  id: serial('id').primaryKey(),
  name: varchar('name', { length: 255 }),
  price: real('price'),
  metadata: json('metadata'),
  createdAt: timestamp('createdAt').defaultNow(),
});

export type Product = InferSelectModel<typeof products>;

export const categories = pgTable('categories', {
  id: serial('id').primaryKey(),

});

export const orders = pgTable('orders', {
  id: serial('id').primaryKey(),
  productId: integer('product_id').references(() => products.id),
  userId: integer('user_id'),
  quantity: integer('quantity'),
  createdAt: timestamp('createdAt').defaultNow(),
});

// queries.ts
import { db } from './db';
import { products, orders } from './schema';
import { eq, sql } from 'drizzle-orm';

export async function getProductWithOrders(productId: number) {
  const product = await db.select().from(products).where(eq(products.id, productId));
  const productOrders = await db
    .select()
    .from(orders)
    .where(eq(orders.productId, productId));
  return { product: product[0], orders: productOrders };
}

export async function getAllProducts() {
  return await db.select().from(products);
}

export async function getProductsByPage(page: number) {
  return await db.select().from(products).limit(20).offset(page * 20);
}

export async function transferStock(fromId: number, toId: number, qty: number) {
  await db.update(products).set({ price: sql`price - ${qty}` }).where(eq(products.id, fromId));
  await db.update(products).set({ price: sql`price + ${qty}` }).where(eq(products.id, toId));
}

export async function searchProducts(userInput: string) {
  return await db.execute(`SELECT * FROM products WHERE name LIKE '%${userInput}%'`);
}

// serverless handler
export async function handler(event: any) {
  const db = drizzle(new Pool({ connectionString: process.env.DB_URL }));
  const products = await db.select().from(products);
  return { body: JSON.stringify(products) };
}
