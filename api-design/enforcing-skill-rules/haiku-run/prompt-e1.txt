function validateOrder(order: Order) {
  if (!order.items.length) throw new Error("Items required");
  if (!order.shippingAddress.city) throw new Error("City required");
}

function calculateTotal(order: Order) {
  validateOrder(order);
  return order.items.reduce((sum, i) => sum + i.qty * 10, 0);
}
