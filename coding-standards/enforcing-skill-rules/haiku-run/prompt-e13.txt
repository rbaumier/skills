export async function signupUser(db: Database, mailer: Mailer, email: string, password: string) {
  if (!validateEmail(email)) throw new Error('invalid email');
  const hashed = await hashPassword(password);
  const welcome = formatWelcomeMessage(email);
  await db.insert('users', { email, password: hashed });
  await mailer.send(email, 'Welcome', welcome);

}
