// Process orders
export async function handleOrderData(userId, items, isUrgent, sendEmail, emailTo, taxRate, discountCode) {
  // Get user
  const u = await db.query('SELECT * FROM users WHERE id = $1', [userId]);
  if (!u.rows[0]) throw new Error('not found');
  const usr = u.rows[0];
  if (usr.blocked) throw new Error('blocked');

  // Process items
  let total = 0;
  let ct = 0;
  for (let i = 0; i < items.length; i++) {
    const itm = items[i];
    if (itm.qty > 0 && itm.active && !usr.blocked) {
      const amt = itm.qty * itm.price * (taxRate ? 1 + taxRate : 1) - (discountCode === 'FREE50' ? 50 : discountCode === 'FREE20' ? 20 : 0);
      if (amt > 0) {
        total += amt;
        ct++;
        // Check stock
        const s = await db.query('SELECT qty FROM stock WHERE id = $1', [itm.id]);
        if (s.rows[0]?.qty < itm.qty) throw new Error('stock');
      }
    }
  }

  if (total > 1000) {
    // Notify
    await mailer.send(usr.email, `Big order: ${total}`);
    await db.query('INSERT INTO big_orders (user_id, total, ts) VALUES ($1, $2, $3)', [userId, total, Date.now()]);
  }

  // Create order
  const order = await db.query('INSERT INTO orders (user_id, total, item_count, status) VALUES ($1, $2, $3, $4) RETURNING *', [userId, total, ct, 'pending']);

  if (isUrgent) {
    await db.query('UPDATE orders SET priority = $1 WHERE id = $2', ['high', order.rows[0].id]);
    await mailer.send(usr.email, 'Urgent order placed');
  } else {
    await db.query('UPDATE orders SET priority = $1 WHERE id = $2', ['normal', order.rows[0].id]);
  }

  if (sendEmail) await mailer.send(emailTo, 'Order confirmation', JSON.stringify(order.rows[0]));

  // Log inside business logic
  console.log('Order created:', order.rows[0].id);
  console.log('Total:', total);

  // Update cache
  cache.set(`user:${userId}:lastOrder`, order.rows[0].id);
  return order.rows[0];
}

export async function removeCartItem(userId: string, itemId: string) {
  const cart = await db.query('SELECT * FROM carts WHERE user_id = $1', [userId]);
  if (!cart.rows[0]) throw new Error('cart not found');
  const item = cart.rows[0].items.find((i: any) => i.id === itemId);
  if (!item) throw new Error('item not in cart');
  await db.query('DELETE FROM cart_items WHERE cart_id = $1 AND item_id = $2', [cart.rows[0].id, itemId]);
}

function validateOrder(order: any) {
  const addr = order.customer.address.city;
  if (!addr) throw new Error('missing city');
  const total = order.items.reduce((s: number, i: any) => s + i.price * i.qty, 0);
  if (total <= 0) throw new Error('invalid total');
  const formatted = `${order.customer.name} - ${total.toFixed(2)} ${order.currency}`;
  return formatted;
}
