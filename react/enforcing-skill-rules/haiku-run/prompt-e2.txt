  const [query, setQuery] = useState("");
  const filtered = products.filter(p => p.name.includes(query));
  return (
    <div>
      <input value={query} onChange={e => setQuery(e.target.value)} />
      {filtered.map(p => <ProductCard key={p.id} product={p} />)}
    </div>
  );
}
